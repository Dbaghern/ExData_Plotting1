#loading in file and changing formats
epc = read.table(file.choose(), sep=";", header=TRUE)
epc$Date = as.Date(epc$Date, "%d/%m/%Y")
epc$Time = strptime(epc$Time, format="%H:%M:%S")

#Creating start and ending dates
startDate = as.Date("01/02/2007", format = "%d/%m/%Y")
endDate = as.Date("02/02/2007", format = "%d/%m/%Y")

#subsetting data
epc.data = subset(epc, epc$Date >= startDate)
epc.data = subset(epc.data, epc.data$Date <= endDate)
